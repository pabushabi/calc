	* "^" - Shift + 6
	* "%" - Shift + 5
	* "(" - Shift + 9
	* ")" - Shift + 0
	* "cos" - c
	* "sin" - s
	* "tan" - t
	* "ln" - l
	* "sqrt" - q
	* copy result to clipboard - Ctrl + c
	* paste from clipboard - Ctrl + v
	* calculate the result - tab
	* erase - backspace
